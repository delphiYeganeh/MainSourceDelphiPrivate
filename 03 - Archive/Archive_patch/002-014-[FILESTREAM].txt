IF NOT EXISTS (SELECT 1 FROM sys.filegroups WHERE [type] = 'FD')
BEGIN        
	DECLARE @DB_Name VARCHAR(100)
	DECLARE @Stsm NVARCHAR(2000)
	SELECT @DB_Name = DB_NAME()
	SET @Stsm = N'ALTER DATABASE [' + @DB_Name + N'] ADD FILEGROUP FILESTREAM_grp CONTAINS FILESTREAM'

	EXECUTE sp_executesql @Stsm
END
GO
