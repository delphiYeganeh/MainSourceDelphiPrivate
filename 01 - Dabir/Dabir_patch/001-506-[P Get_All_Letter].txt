    SELECT DISTINCT letterid FROM #s

            SELECT ArchivePlace FROM #s WHERE letterid=@LID    
            OPEN crin 
            FETCH NEXT FROM crin INTO @ArchPlace    
            WHILE @@FETCH_STATUS = 0

                SET @AllArchPlace=@AllArchPlace+' , '+@ArchPlace
                FETCH NEXT FROM crin INTO @ArchPlace
            END
            CLOSE crin
            DEALLOCATE crin

        FETCH NEXT FROM cr INTO @LID
    END
    CLOSE cr
