BEGIN TRANSACTION
GO
IF EXISTS ( SELECT 1
            FROM   sysobjects 
            WHERE  id = object_id(N'[dbo].[GET_REMAINOFLOAN_BYLOANID]')
                   and OBJECTPROPERTY(id, N'IsProcedure') = 1 )
BEGIN
    DROP PROCEDURE [dbo].[GET_REMAINOFLOAN_BYLOANID]
END
GO

CREATE  procedure [dbo].[GET_REMAINOFLOAN_BYLOANID]
	@LOANID INT, 
	@REMAINOFLOAN BIGINT OUTPUT
AS
	SELECT    @REMAINOFLOAN= SUM(AMOUNT)
		FROM FORCEPAYMENT 
		WHERE  PARENTID=@LOANID AND  FORCETYPEID=1 AND PAID=1
GO

COMMIT
GO
